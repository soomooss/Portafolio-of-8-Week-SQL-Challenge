# 🏦 Case Study #4 - Data Bank
URL https://8weeksqlchallenge.com/case-study-4/

## 📌 Business Overview
Data Bank is a secure, distributed digital bank that operates on a unique cloud infrastructure model: **$1 USD in a customer's account equals exactly 1 MB of free data storage** hosted on the bank's decentralized servers. This project focuses on analyzing node optimization, processing financial transaction streams, and simulating future data storage requirements under multiple financial engineering and interest compounding hypotheses.

### 2. What is the average total historical deposit counts and amounts for all customers?
```sql
WITH ctee AS (
	SELECT
		customer_id,
		COUNT(txn_type) AS count_per_customer,
		SUM(txn_amount) AS amount_per_customer
	FROM data_bank.customer_transactions
	WHERE txn_type = 'deposit'
	GROUP BY customer_id
	ORDER BY customer_id
)
SELECT
	ROUND(AVG(count_per_customer), 2) AS avg_deposits,
	ROUND(AVG(amount_per_customer),2) AS avg_amount
FROM ctee;
```
**Results:**
| avg_deposits | avg_amount |
|--------------|------------|
| 5.34         | 2718.34    |

### 3. For each month - how many Data Bank customers make more than 1 deposit and either 1 purchase or 1 withdrawal in a single month?
```sql
WITH ctee AS (
	SELECT
		customer_id,
		DATE_TRUNC('month', txn_date)::DATE AS txn_month,
		SUM(CASE WHEN txn_type = 'deposit' THEN 1 ELSE NULL END) as deposit_count,
		SUM(CASE WHEN txn_type = 'purchase' THEN 1 ELSE NULL END) as purchase_count,
		SUM(CASE WHEN txn_type = 'withdrawal' THEN 1 ELSE NULL END) as withdrawal_count
	FROM data_bank.customer_transactions
	GROUP BY 1,2 
	ORDER BY customer_id, 2
)
SELECT 
	txn_month,
	COUNT(customer_id)
FROM ctee
WHERE deposit_count > 1 AND (purchase_count >= 1 OR withdrawal_count >= 1)
GROUP BY txn_month
ORDER BY txn_month;
```
**Results:**

|------------|-------|
| 2020-01-01 | 168   |
| 2020-02-01 | 181   |
| 2020-03-01 | 192   |
| 2020-04-01 | 70    |

### 4. What is the closing balance for each customer at the end of the month?
```sql
WITH monthly_changes AS (
	SELECT
		customer_id,
		DATE_TRUNC( 'month', txn_date)::DATE AS txn_month,
		SUM(CASE 
				WHEN txn_type = 'deposit' THEN txn_amount 
				WHEN txn_type = 'purchase'THEN txn_amount * -1
				WHEN txn_type = 'withdrawal' THEN txn_amount * -1
				ELSE 0
			END) AS month_amount
	FROM data_bank.customer_transactions
	GROUP BY 1,2
),
month_range AS(
	SELECT 
		DISTINCT ct.customer_id,
		m.months::DATE as txn_month
	FROM data_bank.customer_transactions AS ct
	CROSS JOIN GENERATE_SERIES('2020-01-01'::DATE,'2020-04-01'::DATE,'1 month' ) AS m(months)
)

SELECT 
	mr.customer_id,
	mr.txn_month,
	COALESCE(mc.month_amount,0) AS month_amount,
	SUM(COALESCE(mc.month_amount,0))OVER (PARTITION BY mr.customer_id ORDER BY mr. txn_month) AS month_balance
FROM month_range AS mr
LEFT JOIN monthly_changes AS mc
	ON mr.customer_id = mc.customer_id AND  mr.txn_month = mc.txn_month
ORDER BY mr.customer_id, mr.txn_month;
```

### 5. What is the percentage of customers who increase their closing balance by more than 5%?
```sql
WITH monthly_changes AS (
	SELECT
		customer_id,
		DATE_TRUNC( 'month', txn_date)::DATE AS txn_month,
		SUM(CASE 
				WHEN txn_type = 'deposit' THEN txn_amount 
				WHEN txn_type = 'purchase'THEN txn_amount * -1
				WHEN txn_type = 'withdrawal' THEN txn_amount * -1
				ELSE 0
			END) AS month_amount
	FROM data_bank.customer_transactions
	GROUP BY 1,2
),
month_range AS(
	SELECT 
		DISTINCT ct.customer_id,
		m.months::DATE as txn_month
	FROM data_bank.customer_transactions AS ct
	CROSS JOIN GENERATE_SERIES('2020-01-01'::DATE,'2020-04-01'::DATE,'1 month' ) AS m(months)
),
months_balances AS (
	SELECT 
		mr.customer_id,
		mr.txn_month,
		COALESCE(mc.month_amount,0) AS month_amount,
		SUM(COALESCE(mc.month_amount,0))OVER (PARTITION BY mr.customer_id ORDER BY mr. txn_month) AS month_balance
	FROM month_range AS mr
	LEFT JOIN monthly_changes AS mc
		ON mr.customer_id = mc.customer_id AND  mr.txn_month = mc.txn_month
),
compare_balance AS (
	SELECT
		customer_id,
		txn_month,
		month_balance,
		COALESCE (LAG(month_balance)OVER(PARTITION BY customer_id ORDER BY txn_month),0) AS last_balance
	FROM months_balances
)

SELECT 
	COUNT(DISTINCT customer_id) FILTER(
		WHERE last_balance > 0 AND ((month_balance - last_balance)* 100.0 / last_balance) > 5.0) AS clientes_con_crecimiento,
	ROUND (
		(COUNT(DISTINCT customer_id) FILTER(
			WHERE last_balance > 0 AND ((month_balance - last_balance)* 100.0 / last_balance) > 5.0)*100.0) /
		(SELECT COUNT(DISTINCT customer_id) FROM data_bank.customer_transactions)
		,1) AS porcentaje 
FROM compare_balance;
```

## D. Extra Challenge
Data Bank wants to try another option which is a bit more difficult to implement - they want to calculate data growth using an interest calculation, just like in a traditional savings account you might have with a bank.
If the annual interest rate is set at 6% and the Data Bank team wants to reward its customers by increasing their data allocation based off the interest calculated on a daily basis at the end of each day, how much data would be required for this option on a monthly basis?
Special notes:
Data Bank wants an initial calculation which does not allow for compounding interest, 
however they may also be interested in a daily compounding interest calculation so you can try to perform this calculation if you have the stamina!

### Interes simple
```sql
WITH saldos_base AS (
	SELECT
		customer_id,
		txn_date,
		SUM(SUM(CASE 
				WHEN txn_type = 'deposit' THEN txn_amount 
				WHEN txn_type = 'purchase'THEN txn_amount * -1
				WHEN txn_type = 'withdrawal' THEN txn_amount * -1
				ELSE 0
			END)) OVER(PARTITION BY customer_id ORDER BY txn_date) AS running_balance
	FROM data_bank.customer_transactions
	GROUP BY customer_id, txn_date
),
calendario AS (
	SELECT 
		DISTINCT customer_id,
		d.dia::DATE AS txn_date
	FROM saldos_base
	CROSS JOIN GENERATE_SERIES('2020-01-01'::DATE, '2020-04-30', '1 day') AS d(dia)
),
bloques AS (
	SELECT
		c.customer_id,
		c.txn_date,
		s.running_balance,
		COUNT(s.running_balance) OVER (PARTITION BY c.customer_id ORDER BY c.txn_date ) AS grupos
	FROM calendario AS c
	LEFT JOIN saldos_base AS s
		ON c.customer_id = s.customer_id AND c.txn_date = s.txn_date
	ORDER BY c.customer_id
),
saldos_diarios AS (
	SELECT 
		customer_id,
		txn_date,
		COALESCE(MAX(running_balance)OVER (PARTITION BY customer_id, grupos),0) AS saldo_diario
	FROM bloques
)
SELECT
	DATE_TRUNC('month', txn_date)::DATE,
	ROUND(SUM(CASE WHEN saldo_diario = 0 OR saldo_diario < 0 THEN 0 ELSE saldo_diario + (saldo_diario * 0.06/365.0) END),1) AS MB_needed
FROM saldos_diarios
GROUP BY 1
ORDER BY 1;
```

| 2020-01-01 | 4146551.5 |
| 2020-02-01 | 7589679.4 |

| 2020-04-01 | 7837884.2 |

### Interes compuesto
```sql
WITH monthly_changes AS (
	SELECT
		customer_id,
		DATE_TRUNC( 'month', txn_date)::DATE AS txn_month,
		SUM(CASE 
				WHEN txn_type = 'deposit' THEN txn_amount 
				WHEN txn_type = 'purchase'THEN txn_amount * -1
				WHEN txn_type = 'withdrawal' THEN txn_amount * -1
				ELSE 0
			END) AS month_amount
	FROM data_bank.customer_transactions
	GROUP BY 1,2
),
month_range AS(
	SELECT 
		DISTINCT ct.customer_id,
		m.months::DATE as txn_month,
		EXTRACT(DAY FROM (m.months + INTERVAL '1 month' - INTERVAL '1 day')) AS dias_mes
	FROM data_bank.customer_transactions AS ct
	CROSS JOIN GENERATE_SERIES('2020-01-01'::DATE,'2020-04-01'::DATE,'1 month' ) AS m(months)
),
m_balance AS (
	SELECT 
		mr.customer_id,
		mr.txn_month,
		mr.dias_mes,
		SUM(COALESCE(mc.month_amount,0))OVER (PARTITION BY mr.customer_id ORDER BY mr. txn_month) AS month_balance
	FROM month_range AS mr
	LEFT JOIN monthly_changes AS mc
		ON mr.customer_id = mc.customer_id AND  mr.txn_month = mc.txn_month
),
interes AS (
	SELECT
		customer_id,
		txn_month,
		month_balance * POWER((1+0.06/365.0),dias_mes) AS saldo_interes
	FROM m_balance
)
SELECT
	txn_month,
	ROUND(SUM(CASE WHEN saldo_interes = 0 OR saldo_interes < 0 THEN 0 ELSE saldo_interes END),2) AS mb_needed
FROM interes
GROUP BY 1
ORDER BY 1;
```

| txn_month  | mb_needed |
|------------|-----------|
| 2020-01-01 | 236798.53 |
| 2020-02-01 | 262757.51 |
| 2020-03-01 | 262304.16 |
| 2020-04-01 | 266166.26 |
